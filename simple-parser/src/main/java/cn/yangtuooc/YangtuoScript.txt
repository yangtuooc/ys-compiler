# 语法规则
program: statement+;

statement
  : dcl  # declaration statement
  | expr  # expression statement
  | assign  # assignment statement
  ;

dcl
  : 'int' Id ( '=' additive )? ';'
  ;

expr
  : additive ';'
  ;

assign
  : Id '=' expr ';'
  ;

additive
  : multiplicative
  | additive plus additive

multiplicative
  : primary
  | multiplicative star multiplicative

primary
  : Id
  | IntLiteral
  | '(' additive ')'
  ;



# 词法规则
Id: [a-zA-Z_][a-zA-Z_]*[0-9]*
IntLiteral: [0-9]+
Plus: '+'
Star: '*'
